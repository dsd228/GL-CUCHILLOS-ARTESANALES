Status: ${response.status}`); return await response.json(); } catch (error) { console.error(`Failed to fetch ${url}: ${error}`); return []; } } // Helper function to load and merge multiple JSON sources async function loadCatalogMultiJson(...urls) { const results = await Promise.all(urls.map(fetchJsonSafe)); return results.flat(); } // Original logic modified to load multiple JSON async function loadProducts() { const productList = await loadCatalogMultiJson( 'products.json', 'cocina.json', 'caza.json', 'parrilla.json', 'diseno.json', 'tradicion.json' ); renderProducts(productList); } loadProducts(); // Other existing rendering logic is preserved...
